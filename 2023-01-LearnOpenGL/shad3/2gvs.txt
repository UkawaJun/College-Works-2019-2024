#version 330 core
layout (location = 0) in vec3 aPos;

out vec3 loca;
uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;
void main()
{
    
    gl_Position = projection * view * model * vec4(aPos, 1.0f);
    loca = aPos;
    //loca = vec3(gl_Position.x,gl_Position.y,gl_Position.z);
}
