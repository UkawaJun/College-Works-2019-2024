#version 330 core
out vec4 FragColor;

in vec3 ourColor;
in vec2 TexCoord;

// texture samplers
uniform sampler2D texture1;
uniform sampler2D texture2;
uniform bool tex;
void main()
{
	// linearly interpolate between both textures (80% container, 20% awesomeface)
	if(tex) FragColor = mix(texture(texture1,vec2(4.0*(1.0-TexCoord.x),4.0*TexCoord.y)), texture(texture2, TexCoord), 0.8);
	else FragColor = vec4(ourColor,1.0)*texture(texture1, TexCoord);
}
